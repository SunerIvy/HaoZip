00041     template <class TBitDecoder>
00042     uint32_t DecodeSymbol(TBitDecoder* bitStream)
00043     {
00044         RCZIP_ASSERT(bitStream != NULL ) ;
00045         int32_t numBits = 0 ;
00046         uint32_t value = bitStream->GetValue(kNumBitsMax);
00047         if (value < m_limits[s_kNumTableBits])
00048         {
00049             numBits = m_lengths[value >> (kNumBitsMax - s_kNumTableBits)];
00050         }
00051         else
00052         {
00053             for (numBits = s_kNumTableBits + 1; value >= m_limits[numBits]; numBits++)
00054             {
00055             }
00056         }
00057         bitStream->MovePos(numBits);
00058         uint32_t index = m_positions[numBits] +
00059                          ((value - m_limits[numBits - 1]) >> (kNumBitsMax - numBits)) ;
00060         if (index >= m_NumSymbols)
00061         {
00062             return 0xFFFFFFFF;
00063         }
00064         return m_symbols[index] ;
00065     }
00066     
00067 private:
00068     
00069     /** m_limits[i] = value limit for symbols with length = i
00070     */
00071     uint32_t m_limits[kNumBitsMax + 1];
00072     
00073     /** m_positions[i] = index in m_symbols[] of first symbol with length = i
00074     */
00075     uint32_t m_positions[kNumBitsMax + 1];
00076     
00077     /** symbol value
00078     */
00079     uint32_t m_symbols[m_NumSymbols];
00080     
00081     /** Table oh length for short codes.
00082     */
00083     byte_t m_lengths[1 << s_kNumTableBits];
