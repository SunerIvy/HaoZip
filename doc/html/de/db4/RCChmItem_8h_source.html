00149     {
00150         if (blockIndex >= m_resetOffsets.size())

00156         if (blockIndex + numBlocks >= m_resetOffsets.size())
00157         {

00162             size = m_resetOffsets[(int32_t)(blockIndex + numBlocks)] - startPos;
00163         }
00164         return true;
