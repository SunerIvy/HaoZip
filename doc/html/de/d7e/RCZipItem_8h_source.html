00089     {
00090         if (sb.m_id != RCZipHeader::NExtraID::kWzAES)

00092             return false;

00096         {
00097             return false;

00103         {
00104             return false;
00105         }
00106 
00107         m_strength = p[4];
00108         m_method = (((uint16_t)p[6]) << 16) | p[5];
00109 
00110         return true;
